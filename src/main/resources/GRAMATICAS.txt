######## INFIXADA

<expr> ::= <term> <expr_tail>
<expr_tail> ::= '+' <term> <expr_tail>
        | '-' <term> <expr_tail>
        | ε
<term> ::= <factor> <term_tail>
<term_tail> ::= '*' <factor> <term_tail>
        | '/' <factor> <term_tail>
        | ε
<factor> ::= <digit>
        | '(' <expr> ')'
<digit> ::= 0 | 1 | ... | 9

######## PREFIXADA

<head> ::= <op> <term> <term>
<term> ::= <dig>
        | <head>
<digit> ::= 0 | 1 | ... | 9
<op> ::= + - * /
